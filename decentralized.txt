\subsection {Problem and Definition of Requirement (Problem ve Gerekliliğin Tanımı)} 

\begin{frame}
\frametitle{Problem and Definition of Requirement (Problem ve Gerekliliğin Tanımı)}
\begin{itemize}
\item Bahis oyunlarında, merkeze bağımlılık sisteme olan güveni azaltıyor.
\item Bahis Siteleri/Kuruluşları birer iş yeri gibi çalışır. Diğer iş yerleri gibi kendi karlarını garanti edebilcek şekilde tasarlanmış bir iş modelleri vardır.
\\"House Always Wins"
\item İnsanlar kazandıklarını düşünmelerine rağmen kazanan her zaman kasa oluyor. 
\item Insanların oynadıkları bahislerde kazanmaları adına hiçbir etkileri yok.
\item Bu yüzden insanların da oyunda etkisi olması adına, kendi NFT varlıkları ile bahislere katılma imkanı sunuyoruz. Bunu, bahislerdeki şans faktörünü koruyarak gerçekleştiriyoruz.

\end{itemize}
\end{frame}

\subsection {Solution Proposal - Model} 

\begin{frame}
\frametitle{Solution Proposal - Model}
\begin{itemize}
\item Different user types of the system must be specified in the model
\item Basic functions should be displayed on the figure
\item Arrows should show the direction of data flow (with data type)
\item I recommend using Draw.io or alike system (with colour selection for the colour blind people)
\end{itemize}

\end{frame}

\subsection {Original Value (Özgün Değer)} 

\begin{frame}
\frametitle{Original Value (Özgün Değer)}
\begin{itemize}
\item Bahis oyunlarından ayıran özellik, kendi varlıklarımıza sahip olmamız ve bunlarla bahislere katılabilmemiz.
\item NFT varlıklarımızı belirli koşullar dahilinde geliştirebiliyoruz.
\item Böylelikle oyunculara, bahis oyunu tarafında stratejik çeşitlilikler sunarak kazanmak için farklı opsiyonlar veriyoruz. 
\end{itemize}
\end{frame}

\subsection {Widespread Impact (Yaygın Etki)} 

\begin{frame}
\frametitle{Widespread Impact (Yaygın Etki)}
\begin{itemize}
\item İnsanların bahis oyunlarına karşı tutumlarını ve bakış açılarını değiştirmeyi amaçlıyoruz.
\item İnsanların bahis oynamak için daha güvenli platformlar aramasından doğan ihtiyaçları gören bu sektördeki şirketlerin gereken hamleleri yapmasını planlıyoruz
\end{itemize}
\end{frame}
